なぜスマートポインタを使うのか

スマートポインタとは、deleteを自動でやってくれる仕組み
	(テンプレートの仕組みの一つにより自動で解放される)

//スコープから外れると自動で解放してくれるクラス
template <typename T>
class ScopedPtr{
private:
	T* ptr_;

public:
	ScopedPtr(T* ptr) :
		ptr_(ptr){
			
		}
	~ScopedPtr(){
		delete ptr_;
	}
	T* operator->(){
		return ptr_;
	}
}

int main()
{
	{
		ScopedPtr<Mahoshojo> maho(new Mahoshojo());
	}
}
